### 1.2 Endpoint 设计

```python
@app.get(
    "/api/v1/clients/{client_id}/fact-bundle",
    response_model=ClientFactBundleResponse,
)
def get_client_fact_bundle_api(
    client_id: str,
    include_archived: bool = Query(False),
    lite: bool = Query(False, description="只返回 counts, 不拿事实列表"),
) -> ClientFactBundleResponse:
    """Phase 1 F1.4a · 暴露 ClientFactView 给前端 / 机器人 / 任何 caller"""
    from app.modules.client import get_client_fact_view
    view = get_client_fact_view(state.db)
    bundle = (
        view.get_fact_bundle_lite(client_id)
        if lite
        else view.get_fact_bundle(client_id, include_archived=include_archived)
    )
    if bundle is None:
        raise HTTPException(404, "Client not found or archived")
    return _bundle_to_response(bundle)
```
