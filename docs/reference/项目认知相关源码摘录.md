   309	async function requestForm<T>(path: string, formData: FormData, options?: FormRequestOptions): Promise<T> {
   310	  const onProgress = options?.onProgress;
   311	  if (onProgress) {
   312	    return new Promise<T>((resolve, reject) => {
   313	      const xhr = new XMLHttpRequest();
   314	      xhr.open((options?.method || 'POST').toUpperCase(), `${baseUrl}${path}`);
   315	      const headers = new Headers(options?.headers || {});
   316	      headers.forEach((value, key) => {
   317	        xhr.setRequestHeader(key, value);
   318	      });
   319	      xhr.upload.onprogress = (event) => {
   320	        onProgress(event.loaded, event.lengthComputable ? event.total : 0);
   321	      };
   322	      xhr.onerror = () => {
   323	        reject(new Error('附件上传失败，请稍后重试。'));
   324	      };
   325	      xhr.onload = () => {
   326	        const text = xhr.responseText || '';
   327	        if (xhr.status < 200 || xhr.status >= 300) {
   328	          let detail = text;
   329	          try {
   330	            const payload = JSON.parse(text) as { detail?: string };
   331	            detail = payload.detail || text;
   332	          } catch {}
   333	          reject(new Error(detail || `HTTP ${xhr.status}`));
   334	          return;
   335	        }
   336	        try {
   337	          resolve(JSON.parse(text) as T);
   338	        } catch (error) {
   339	          reject(error instanceof Error ? error : new Error('附件上传响应解析失败'));
   340	        }
   341	      };
   342	      xhr.send(formData);
