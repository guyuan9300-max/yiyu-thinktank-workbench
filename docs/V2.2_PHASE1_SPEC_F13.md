## 0 · F1.3 一句话目标

**建立 `ClientScopeFilter`:所有 client_id 写入路径走它做 scope check + frozen check + archived check,杜绝 v1.0 "frozen 状态被云端覆盖" 这类 bug。**

完成标志:`F1.4` 切迁时, 所有 `INSERT INTO ... VALUES (..., client_id, ...)` 之前都先调 `scope.assert_writable(client_id)`。

## 1 · 设计原则

### 1.1 三个核心方法 (按使用频率排)

```python
class ClientScopeFilter:
    # 1. 写入守门 (最频繁) — 一行 guard, 写入前必调
    def assert_writable(self, client_id: str) -> None:
        """如果 client 不存在 / frozen / archived / lost, raise ScopeFilterError"""

    # 2. 批量过滤 — 查询场景, 把一组 id 过滤成"合规子集"
    def scope_query(
        self, client_ids: list[str] | None, *,
        include_frozen: bool = False, include_archived: bool = False,
        require_writable: bool = False,
    ) -> list[str]:
        """过滤一组 client_id, 返回符合条件的子集 (按 name 排序)"""

    # 3. 记录字典过滤 — list[Record] 按 client_id 已分组的场景
    def filter_records_by_client(
        self, records_by_client: dict[str, T], **scope_kwargs
    ) -> dict[str, T]:
        """过滤掉不在 scope 的 client_id, 保留 in-scope 的"""
```

## 4 · `scope_filter.py` 详细设计

```python
"""ClientScopeFilter · client_id 写入路径统一守门 (Phase 1 F1.3)

服务: V2.2_NORTH_STAR.md 目标 A (现有功能不掉链)
按: docs/V2.2_PHASE1_SPEC_F13.md

关键场景:
1. main.py 写入 task/event_line/commitment 之前: scope.assert_writable(client_id)
2. 批量查询 (按 client_id 过滤): scope.scope_query(client_ids)
3. 多账号同步: F1.7 client_stage_audit + 同步前 scope.assert_writable

依赖: F1.1 ClientRepository (复用 is_frozen / is_archived / can_write)
"""
from __future__ import annotations

from typing import TypeVar

from .repository import ClientRepository, get_client_repository
from .types import ClientStage


T = TypeVar("T")


class ScopeFilterError(Exception):
    """写入被拦截时抛出"""

    def __init__(self, client_id: str, reason: str):
        self.client_id = client_id
        self.reason = reason
        super().__init__(f"client {client_id} 不可写: {reason}")


# reason 常量
REASON_NOT_FOUND = "client_not_found"
REASON_FROZEN = "client_frozen"
REASON_ARCHIVED = "client_archived"
REASON_LOST = "client_lost"


class ClientScopeFilter:
    """所有 client_id 写入/查询的统一守门

    Args:
        client_repo: 复用 F1.1 ClientRepository (推荐注入)
                     若不传则自动 new (需要 db 入参)
    """

    def __init__(self, client_repo: ClientRepository) -> None:
        self._repo = client_repo

    # ── 1. 写入守门 (最高频) ──────────────────────────────────────
    def assert_writable(self, client_id: str) -> None:
        """如果 client 不可写, raise ScopeFilterError. 写入前必调."""
        if not client_id:
            raise ScopeFilterError("", REASON_NOT_FOUND)
        client = self._repo.get_by_id(client_id)
        if client is None:
            raise ScopeFilterError(client_id, REASON_NOT_FOUND)
        # stage 路径
        if client.stage == "frozen":
            raise ScopeFilterError(client_id, REASON_FROZEN)
        if client.stage == "archived":
            raise ScopeFilterError(client_id, REASON_ARCHIVED)
        if client.stage == "lost":
            raise ScopeFilterError(client_id, REASON_LOST)
        # 否则可写

    def is_writable(self, client_id: str) -> bool:
        """不抛异常版本 — 一行 bool check, 用于条件分支"""
        try:
            self.assert_writable(client_id)
            return True
        except ScopeFilterError:
            return False

    # ── 2. 批量查询过滤 ───────────────────────────────────────────
    def scope_query(
        self,
        client_ids: list[str] | None = None,
        *,
        include_frozen: bool = False,
        include_archived: bool = False,
        require_writable: bool = False,
    ) -> list[str]:
        """过滤一组 client_id 返回 in-scope 的子集.

        Args:
            client_ids: None = 全部活跃; 否则按这组过滤
            include_frozen: True 时不过滤 stage='frozen'
            include_archived: True 时不过滤 stage='archived'/'lost'
            require_writable: True 时只返回 can_write() 的 (==> include_frozen=False
                              + include_archived=False, 覆盖其他选项)

        Returns:
            合规 client_id 列表, 按 name 排序
        """
        if require_writable:
            # 强制收紧
            include_frozen = False
            include_archived = False

        # 决定候选池
        if client_ids is None:
            # 没传 ids 时, 用 list_all 拿全部 (排除 archived 默认)
            candidates = self._repo.list_all(include_frozen=include_frozen)
            # list_all 默认不含 archived/lost, include_archived 是个 alias
            # (实际语义: include_frozen 控制 frozen, include_archived 控制 archived+lost)
            if include_archived:
                # 需要补 archived/lost
                all_inc = self._repo.list_all(include_frozen=True)
                candidates_ids_existing = {c.id for c in candidates}
                for c in all_inc:
                    if c.id not in candidates_ids_existing:
                        candidates = list(candidates) + [c]
            ids = [c.id for c in candidates]
        else:
            # 给定 ids, 一个个 check
            ids = []
            for cid in client_ids:
                client = self._repo.get_by_id(cid)
                if client is None:
                    continue
                # stage 过滤
                if client.stage == "frozen" and not include_frozen:
                    continue
                if client.stage in ("archived", "lost") and not include_archived:
                    continue
                ids.append(cid)

        # 排序 (按 name)
        name_map: dict[str, str] = {}
        for cid in ids:
            client = self._repo.get_by_id(cid)
            if client:
                name_map[cid] = client.name
        return sorted(ids, key=lambda c: name_map.get(c, ""))

    # ── 3. 记录字典过滤 ───────────────────────────────────────────
    def filter_records_by_client(
        self,
        records_by_client: dict[str, T],
        *,
        include_frozen: bool = False,
        include_archived: bool = False,
        require_writable: bool = False,
    ) -> dict[str, T]:
        """已经按 client_id 分组的字典 → 过滤掉不在 scope 的 key.

        典型场景: 周复盘聚合按客户分组后, 过滤掉 frozen/archived 客户
        """
        valid_ids = set(
            self.scope_query(
                list(records_by_client.keys()),
                include_frozen=include_frozen,
                include_archived=include_archived,
                require_writable=require_writable,
            )
        )
        return {cid: records_by_client[cid] for cid in valid_ids if cid in records_by_client}


def get_client_scope_filter(db_or_repo) -> ClientScopeFilter:
    """工厂. 接受 db (自动 new repo) 或 ClientRepository (直接复用)."""
    if isinstance(db_or_repo, ClientRepository):
        return ClientScopeFilter(db_or_repo)
    return ClientScopeFilter(get_client_repository(db_or_repo))
```

## 8 · 风险点 + 兜底

| 风险 | 概率 | 兜底 |
|---|---|---|
| stage='待导入资料' 这类自定义 stage 被 ClientStage Literal 限制 | 已发生 | F1.1 的 _row_to_record 已经用 type: ignore 容错。F1.3 也接受任意 stage 字符串 |
| scope_query() 性能问题 (1000+ 客户) | 低 | 当前实现内部 get_by_id 多次查询, 但 list_all 已 ORDER BY name 优化。Phase 1 可接受 |
| require_writable 跟其他 flag 冲突的语义不清晰 | 中 | spec §4 明确: require_writable 覆盖其他 flag (强制收紧) |
| ScopeFilterError 跟现有 HTTPException 不一致 | 中 | F1.4 切迁时, main.py 写 except ScopeFilterError → raise HTTPException(404/403) |
| ClientRepository 注入 vs new 的歧义 | 低 | get_client_scope_filter 工厂同时支持两种, 不强制 |
