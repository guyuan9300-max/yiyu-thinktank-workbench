### 2.4 `repository.py` 详细设计

```python
"""ClientRepository · 客户数据唯一读写入口 (Phase 1 F1.1 完整实现)

跟 W1 OrganizationRepository 模式一致 (modules/organization/repository.py)。
镜像 cloud_backend mobile endpoints 的 ClientRecord shape。

下游消费者:
- main.py 中 ~127 处 clients/client_folders/client_units 直查 SQL → Phase 1 F1.4 切走 top 50 处
- ClientFactView (F1.2) → 用本 Repository 拿 base client + 合并 event_lines/tasks/...
- ClientScopeFilter (F1.3) → 用 is_frozen/is_archived 做仲裁
"""
from __future__ import annotations
import json
import sqlite3
import uuid
from datetime import datetime, timezone
from typing import Any, Protocol

from .types import (
    ClientRecord, ClientCreatePayload, ClientUpdatePayload, ClientStage,
)


# ── DB 协议 + Adapter (跟 W3 模式一致) ──
class _DbLike(Protocol):
    def fetchone(self, query: str, params: tuple = ()) -> sqlite3.Row | None: ...
    def fetchall(self, query: str, params: tuple = ()) -> list[sqlite3.Row]: ...
    def execute(self, query: str, params: tuple = ()) -> None: ...


class _ConnAdapter:
    def __init__(self, conn: sqlite3.Connection):
        self._conn = conn
    def fetchone(self, query, params=()):
        return self._conn.execute(query, params).fetchone()
    def fetchall(self, query, params=()):
        return self._conn.execute(query, params).fetchall()
    def execute(self, query, params=()):
        self._conn.execute(query, params)


def _wrap_db(db: Any) -> _DbLike:
    if isinstance(db, sqlite3.Connection):
        return _ConnAdapter(db)
    return db


def _now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()


def _row_to_record(row: Any) -> ClientRecord:
    """sqlite3.Row → ClientRecord 不可变"""
    return ClientRecord(
        id=str(row["id"]),
        name=str(row["name"]),
        alias=str(row["alias"] or ""),
        domain=str(row["domain"] or ""),
        type=str(row["type"] or ""),
        intro=str(row["intro"] or ""),
        stage=str(row["stage"] or "active"),  # type: ignore
        color=str(row["color"] or "#5B7BFE"),
        created_at=str(row["created_at"]),
        updated_at=str(row["updated_at"]),
    )


class ClientRepository:
    """客户数据 SSOT"""

    def __init__(self, db: Any):
        self._db = _wrap_db(db)

    # ── 读 · 基础 ──
    def get_by_id(self, client_id: str) -> ClientRecord | None:
        if not client_id:
            return None
        row = self._db.fetchone(
            "SELECT * FROM clients WHERE id = ?", (client_id,)
        )
        return _row_to_record(row) if row else None

    def get_by_alias(self, alias: str) -> ClientRecord | None:
        if not alias:
            return None
        row = self._db.fetchone(
            "SELECT * FROM clients WHERE alias = ? LIMIT 1", (alias,)
        )
        return _row_to_record(row) if row else None

    def list_active(self) -> list[ClientRecord]:
        """通过 v_active_clients view 拿所有未冻结 client.

        W1 已经建了 v_active_clients view (排除 stage IN ('frozen','archived','lost'))
        """
        rows = self._db.fetchall(
            "SELECT * FROM clients WHERE id IN (SELECT id FROM v_active_clients) "
            "ORDER BY name"
        )
        return [_row_to_record(r) for r in rows]

    def list_all(self, include_frozen: bool = False) -> list[ClientRecord]:
        if include_frozen:
            rows = self._db.fetchall("SELECT * FROM clients ORDER BY name")
        else:
            rows = self._db.fetchall(
                "SELECT * FROM clients WHERE stage NOT IN ('archived', 'lost') "
                "ORDER BY name"
            )
        return [_row_to_record(r) for r in rows]

    # 兼容 W2 占位接口 (其他代码可能已经在调)
    def list_active_client_ids(self) -> list[str]:
        return [c.id for c in self.list_active()]

    def count_active(self) -> int:
        return len(self.list_active())

    # ── 写 · CRUD ──
    def create(self, payload: ClientCreatePayload) -> ClientRecord:
        if not payload.name.strip():
            raise ValueError("name 不能为空")
        timestamp = _now_iso()
        client_id = f"client_{uuid.uuid4().hex[:10]}"
        self._db.execute(
            """
            INSERT INTO clients (
                id, name, alias, domain, type, intro, stage, color,
                created_at, updated_at
            ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            """,
            (
                client_id,
                payload.name.strip(),
                payload.alias.strip() or payload.name.strip(),
                payload.domain.strip(),
                payload.type.strip(),
                payload.intro.strip(),
                payload.stage,
                payload.color,
                timestamp,
                timestamp,
            ),
        )
        result = self.get_by_id(client_id)
        assert result is not None, "create 后立即读不到,db 异常"
        return result

    def update(self, client_id: str, patch: ClientUpdatePayload) -> ClientRecord:
        existing = self.get_by_id(client_id)
        if existing is None:
            raise ValueError(f"client not found: {client_id}")
        # 构建 dynamic update
        updates: list[str] = []
        params: list[Any] = []
        for field_name in ("name", "alias", "domain", "type", "intro", "stage", "color"):
            new_value = getattr(patch, field_name, None)
            if new_value is not None:
                updates.append(f"{field_name} = ?")
                params.append(new_value)
        if not updates:
            return existing
        updates.append("updated_at = ?")
        params.append(_now_iso())
        params.append(client_id)
        self._db.execute(
            f"UPDATE clients SET {', '.join(updates)} WHERE id = ?",
            tuple(params),
        )
        result = self.get_by_id(client_id)
        assert result is not None
        return result

    # ── 状态唯一真相 (L2 核心) ──
    def freeze(self, client_id: str, reason: str = "") -> ClientRecord:
        """冻结 = 不被云端同步覆盖。是 L2 状态字段唯一真相的入口。

        修 v1.0 'frozen_at 被云端覆盖' bug 的关键 — 只能通过本方法改 stage。
        """
        existing = self.get_by_id(client_id)
        if existing is None:
            raise ValueError(f"client not found: {client_id}")
        if existing.stage == "frozen":
            return existing  # 已冻结
        # TODO Phase 1 F1.7:写入 frozen_audit_log(reason, frozen_at, by_user)
        return self.update(client_id, ClientUpdatePayload(stage="frozen"))

    def unfreeze(self, client_id: str) -> ClientRecord:
        existing = self.get_by_id(client_id)
        if existing is None:
            raise ValueError(f"client not found: {client_id}")
        return self.update(client_id, ClientUpdatePayload(stage="active"))

    def archive(self, client_id: str) -> ClientRecord:
        return self.update(client_id, ClientUpdatePayload(stage="archived"))

    def is_frozen(self, client_id: str) -> bool:
        client = self.get_by_id(client_id)
        return client is not None and client.stage == "frozen"

    def is_archived(self, client_id: str) -> bool:
        client = self.get_by_id(client_id)
        return client is not None and client.stage in ("archived", "lost")

    def can_write(self, client_id: str) -> bool:
        """是否允许业务写入. frozen + archived 都不允许"""
        client = self.get_by_id(client_id)
        return client is not None and client.stage not in (
            "frozen", "archived", "lost"
        )


def get_client_repository(db: Any) -> ClientRepository:
    return ClientRepository(db)
```

@dataclass(frozen=True)
class ClientFactBundle:
    """一个客户的全部已知事实 — L2 共识层核心数据结构"""

    event_lines: list[EventLineFact]           # 事件线 + 状态

    snapshot_at: str                           # 这份 bundle 的快照时间
    sources: dict[str, str]                    # 每个字段的数据源 (provenance)

    """L2 共识层:所有 client_id 相关的查询走这层"""

        self,

        commitment_repo: CommitmentRepository,
        glossary_repo: GlossaryRepository,
        # event_line / task / dna 暂时直查 db,等后续 phase 再扩 repository

    # ── 核心方法 ──
    def get_fact_bundle(
        self, client_id: str, *, include_archived: bool = False
    ) -> ClientFactBundle | None: ...

    def list_bundles_for_active_clients(self) -> list[ClientFactBundle]: ...

    # ── 细粒度查询 (上层不想拿整个 bundle 时用) ──
    def list_event_lines(self, client_id: str) -> list[EventLineFact]: ...
    def list_tasks(self, client_id: str, *, include_done: bool = False) -> list[TaskFact]: ...
    def list_commitments(self, client_id: str) -> list[CommitmentFact]: ...
    def list_dna_documents(self, client_id: str) -> list[DnaDocumentRef]: ...

    # ── 变更通知 (为 F3 NarrativeKernel 事件驱动准备) ──
    def emit_fact_changed(
        self, client_id: str, fact_type: str, change_kind: str

### 4.2 设计核心

```python
class ClientScopeFilter:
    """所有 client_id 查询走它做 scope check + frozen check"""

    def __init__(self, client_repo: ClientRepository): ...

    def scope_query(
        self,
        client_ids: list[str] | None,
        *,
        include_frozen: bool = False,
        include_archived: bool = False,
        require_writable: bool = False,
    ) -> list[str]:
        """过滤一组 client_id, 返回符合条件的子集.

        Args:
            client_ids: None = 全部活跃; 否则按这组过滤
            include_frozen: True 时不过滤 stage='frozen'
            include_archived: True 时不过滤 stage='archived'/'lost'
            require_writable: True 时只返回 can_write() 的

        Returns:
            合规的 client_id 列表 (按 name 排序)
        """

    def assert_writable(self, client_id: str) -> None:
        """如果 client 不可写,raise ScopeFilterError. 写入前必调."""

    def filter_records_dict(
        self,
        records_by_client: dict[str, Any],
        **scope_kwargs,
    ) -> dict[str, Any]:
        """已经按 client_id 分组的记录字典 → 过滤掉不在 scope 的 client_id"""
```

### 4.3 关键集成点

`backend/app/main.py` 中 client_id 查询调用前一律 `scope = ClientScopeFilter(client_repo); scope.assert_writable(client_id)`。这是 F1.4 切迁的标准模板。

CREATE TABLE client_stage_audit (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    client_id TEXT NOT NULL,
    old_stage TEXT,
    new_stage TEXT,
    changed_by TEXT NOT NULL,           -- 'local' / 'cloud_sync' / 'user:{user_id}'

    FOREIGN KEY(client_id) REFERENCES clients(id) ON DELETE CASCADE

ClientRepository.freeze() / unfreeze() / archive() 三个写入方法都写 audit log。

cloud sync 时不再直接 UPDATE clients.stage,而是:
1. 比较 local stage vs cloud stage
2. 如果 local 是 frozen,**云端不允许覆盖** (这就是 v1.0 bug 的修法)
3. 如果是其他状态变化,记 audit log 后再写
